Solution : 


Code : 
```
class Solution{
    public:
    int findSubString(string str){
        set<char> s;
        for(auto &i : str) s.insert(i);
        int start = 0;
        int end = 0;
        int n = str.length();
        int mini = INT_MAX;
        map<char,int> mp;
        while(start < n && end < n){
            mp[str[end]]++;
            while(mp.size() == s.size()){
                mini = min(mini, end -start + 1);
                mp[str[start]]--;
                if(mp[str[start]] == 0) mp.erase(str[start]);
                start++;
            }
            end++;
        }
        return mini;
    }
};

```

TC : O(n)<br>
SC : O(52)
